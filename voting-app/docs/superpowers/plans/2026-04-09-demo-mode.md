> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a complete demo mode to the FRCA Election App for use at council proposal meetings, including fictional candidate generation, lenient code entry, dual ballot handouts, captive portal support, and a council proposal document.

**Architecture:** Single-file Flask app (`app.py` ~2450 lines) gets demo mode as a settings flag (`is_demo`). PDF generation is extracted from route handlers into a shared module (`pdf_generators.py`) so both routes and CLI scripts can generate PDFs. Name generation lives in `demo_names.py`. Two CLI scripts (`seed_demo.py`, `reset_app.py`) provide headless setup/teardown. Captive portal probe routes redirect phones to the voting page automatically.

**Tech Stack:** Python 3.11+, Flask 3.x, ReportLab 4.x, SQLite3, qrcode 8.x, pytest 8.x. New dependency: `markdown` (for rendering council proposal in-browser).

"""
Dutch name generation for demo mode.

Generates fictional Dutch-sounding candidate names by mashing up surname
components from a real member list, or falls back to a hardcoded list.
"""

import random
import re

DUTCH_FIRST_NAMES = [
    "Pieter", "Hendrik", "Willem", "Gerrit", "Jacobus", "Cornelis",
    "Bastiaan", "Adriaan", "Arend", "Dirk", "Reinier", "Frederik",
    "Johannes", "Maarten", "Sebastiaan", "Theodoor", "Klaas", "Rutger",
    "Albert", "Teunis", "Marinus", "Abraham", "Izaak", "Nicolaas",
]

DUTCH_PARTICLES = ["van", "de", "van der", "ten", "ter", "van den"]

FALLBACK_ELDER_CANDIDATES = [
    "Pieter van Rijksen",
    "Hendrik Brouwerhof",
    "Willem de Kempenaar",
    "Gerrit van Dijkstra",
]

FALLBACK_DEACON_CANDIDATES = [
    "Arend Visserman",
    "Dirk van Leeuwenburg",
    "Reinier Mulderhoek",
    "Frederik Bosveldt",
]

# Common Dutch surname roots for mash-up
_SURNAME_ROOTS_PREFIX = [
    "Brouw", "Dijk", "Kemp", "Molen", "Bos", "Veld", "Berg",
    "Hoek", "Kamp", "Riet", "Sloot", "Vliet", "Woud", "Zand",
    "Leeuw", "Visser", "Mulder", "Bakker", "Smit", "Kuiper",
    "Schoon", "Nieuw", "Groot", "Klein", "Roos", "Bloem",
]

_SURNAME_ROOTS_SUFFIX = [
    "stra", "hof", "man", "burg", "dam", "veld", "hoek", "beek",
    "kamp", "enaar", "huis", "berg", "broek", "dijk", "meer",
    "wal", "boom", "en", "ink", "ema", "sma", "inga",
]


def _extract_surname(full_name):
    """Extract surname from a full name, correctly handling Dutch particles."""
    parts = full_name.strip().split()
    if len(parts) <= 1:
        return None

    # Check for multi-word particles: "van der", "van den"
    for i in range(1, len(parts)):
        prefix = " ".join(parts[1:i+1]).lower()
        if prefix in ("van der", "van den", "van de"):
            return " ".join(parts[1:])
    # Single-word particles
    if parts[1].lower() in ("van", "de", "ten", "ter"):
        return " ".join(parts[1:])
    # No particle
    return " ".join(parts[1:])


def _split_surname_roots(surname):
    """Split a surname into root components for mash-up."""
    # Strip particles for root extraction
    clean = surname
    for p in sorted(DUTCH_PARTICLES, key=len, reverse=True):
        if clean.lower().startswith(p + " "):
            clean = clean[len(p):].strip()
            break

    if len(clean) < 3:
        return []

    # Simple syllable split: try to split roughly in half,
    # at a consonant-vowel boundary
    vowels = set("aeiouAEIOU")
    mid = len(clean) // 2
    best_split = mid

    # Search for a good split point near the middle
    for offset in range(0, len(clean) // 2):
        for pos in [mid + offset, mid - offset]:
            if 2 <= pos < len(clean) - 1:
                if clean[pos - 1] not in vowels and clean[pos] in vowels:
                    best_split = pos
                    break
                if clean[pos] not in vowels and clean[pos - 1] in vowels:
                    best_split = pos
                    break
        else:
            continue
        break

    prefix = clean[:best_split]
    suffix = clean[best_split:]

    roots = []
    if len(prefix) >= 2:
        roots.append(prefix)
    if len(suffix) >= 2:
        roots.append(suffix)
    return roots


def _generate_mashup_names(member_names, count=8):
    """Generate fictional names by mashing up surname components from member list."""
    # Extract all surnames and their roots
    surnames = []
    all_prefix_roots = list(_SURNAME_ROOTS_PREFIX)
    all_suffix_roots = list(_SURNAME_ROOTS_SUFFIX)

    for name in member_names:
        surname = _extract_surname(name)
        if surname:
            surnames.append(surname.lower())
            roots = _split_surname_roots(surname)
            if len(roots) >= 2:
                all_prefix_roots.append(roots[0])
                all_suffix_roots.append(roots[1])
            elif len(roots) == 1:
                all_prefix_roots.append(roots[0])

    # Generate unique fictional names
    member_surnames_lower = set(s.lower() for s in surnames)
    member_full_lower = set(n.strip().lower() for n in member_names)
    used_surnames = set()
    generated = []
    first_names_pool = list(DUTCH_FIRST_NAMES)
    random.shuffle(first_names_pool)
    first_idx = 0

    max_attempts = count * 50
    attempts = 0

    while len(generated) < count and attempts < max_attempts:
        attempts += 1

        # Pick a prefix root and suffix root from different source surnames
        prefix = random.choice(all_prefix_roots)
        suffix = random.choice(all_suffix_roots)

        # Capitalise and combine
        new_surname = prefix.capitalize() + suffix.lower()

        # Optionally add a particle (~40% chance)
        if random.random() < 0.4:
            particle = random.choice(DUTCH_PARTICLES)
            full_surname = f"{particle} {new_surname}"
        else:
            full_surname = new_surname

        # Check uniqueness and safety
        if full_surname.lower() in used_surnames:
            continue
        if full_surname.lower() in member_surnames_lower:
            continue
        if new_surname.lower() in member_surnames_lower:
            continue

        # Pick a first name
        first_name = first_names_pool[first_idx % len(first_names_pool)]
        first_idx += 1

        full_name = f"{first_name} {full_surname}"
        if full_name.lower() in member_full_lower:
            continue

        used_surnames.add(full_surname.lower())
        generated.append(full_name)

    return generated


def generate_demo_names(count=8, member_names=None):
    """
    Generate fictional Dutch candidate names for demo mode.

    If member_names is provided and has enough entries (>= 4), uses mash-up.
    Otherwise returns the hardcoded fallback list.

    Returns a list of `count` unique fictional full names.
    """
    if member_names is None:
        member_names = []

    if len(member_names) >= 4:
        names = _generate_mashup_names(member_names, count=count)
        if len(names) == count:
            return names

    # Fallback
    fallback = FALLBACK_ELDER_CANDIDATES + FALLBACK_DEACON_CANDIDATES
    return fallback[:count]


def load_member_names_from_db(db_path):
    """
    Load member names from the app database.

    Checks the members table for rows with first_name and last_name.
    Returns a list of full name strings.
    """
    import sqlite3
    if not os.path.exists(db_path):
        return []
    try:
        conn = sqlite3.connect(db_path)
        conn.row_factory = sqlite3.Row
        rows = conn.execute("SELECT first_name, last_name FROM members").fetchall()
        conn.close()
        return [f"{r['first_name']} {r['last_name']}" for r in rows if r['first_name'] and r['last_name']]
    except Exception:
        return []


def load_member_names_from_external(data_dir):
    """
    Check external sources for member names in priority order:
    1. data/members.db (SQLite with members/attendees table)
    2. data/members.csv (CSV with name/full_name column)
    3. config/members.txt (plain text, one name per line)

    Returns a list of full name strings, or empty list if none found.
    """
    import sqlite3
    import csv

    # 1. External SQLite
    ext_db = os.path.join(data_dir, "members.db")
    if os.path.exists(ext_db):
        try:
            conn = sqlite3.connect(ext_db)
            conn.row_factory = sqlite3.Row
            # Try 'members' table first, then 'attendees'
            for table in ["members", "attendees"]:
                try:
                    cols = [row[1] for row in conn.execute(f"PRAGMA table_info({table})").fetchall()]
                    if "full_name" in cols:
                        rows = conn.execute(f"SELECT full_name FROM {table}").fetchall()
                        names = [r["full_name"] for r in rows if r["full_name"]]
                    elif "name" in cols:
                        rows = conn.execute(f"SELECT name FROM {table}").fetchall()
                        names = [r["name"] for r in rows if r["name"]]
                    else:
                        continue
                    if names:
                        conn.close()
                        return names
                except Exception:
                    continue
            conn.close()
        except Exception:
            pass

    # 2. CSV
    csv_path = os.path.join(data_dir, "members.csv")
    if os.path.exists(csv_path):
        try:
            with open(csv_path, "r", encoding="utf-8") as f:
                reader = csv.DictReader(f)
                fieldnames_lower = {fn.lower(): fn for fn in (reader.fieldnames or [])}
                col = fieldnames_lower.get("full_name") or fieldnames_lower.get("name")
                if col:
                    return [row[col] for row in reader if row.get(col, "").strip()]
        except Exception:
            pass

    # 3. Plain text
    base_dir = os.path.dirname(data_dir)
    txt_path = os.path.join(base_dir, "config", "members.txt")
    if os.path.exists(txt_path):
        try:
            with open(txt_path, "r", encoding="utf-8") as f:
                return [line.strip() for line in f if line.strip()]
        except Exception:
            pass

    return []


```

def _generate_qr_image(url, size=120):
    """Generate a QR code image as a ReportLab-compatible ImageReader."""
    qr = qrcode.QRCode(
        version=None,
        error_correction=qrcode.constants.ERROR_CORRECT_H,
        box_size=10,
        border=1,
    )
    qr.add_data(url)
    qr.make(fit=True)
    img = qr.make_image(fill_color="black", back_color="white")
    img_buf = io.BytesIO()
    img.save(img_buf, format="PNG")
    img_buf.seek(0)
    return ImageReader(img_buf)

def generate_paper_ballot_pdf(election_name, round_number, office_data,
                              member_count=0, is_demo=False):
    """
    Generate printable paper ballot forms.

    Args:
        election_name: Election name string
        round_number: Current round number
        office_data: List of dicts with 'office' (dict with name, max_selections)
                     and 'candidates' (list of dicts with name)
        member_count: Number of members (for calculating total ballots)
        is_demo: If True, add DEMO watermark

    Returns:
        BytesIO buffer containing the PDF
    """
    buf = io.BytesIO()
    c = canvas.Canvas(buf, pagesize=A4)
    width, height = A4

    total_candidates = sum(len(item["candidates"]) for item in office_data)
    header_height = 14 * mm
    if round_number > 1:
        header_height += 5 * mm
    office_height = len(office_data) * 5 * mm
    candidate_height = total_candidates * 5 * mm
    padding = 6 * mm
    ballot_h = header_height + office_height + candidate_height + padding

    margin = 8 * mm
    col_gap = 6 * mm
    row_gap = 4 * mm
    col_w = (width - 2 * margin - col_gap) / 2
    usable_height = height - 2 * margin
    rows_per_page = max(1, int((usable_height + row_gap) / (ballot_h + row_gap)))
    ballots_per_page = rows_per_page * 2

    total_ballots = max(member_count + 10, 30) if member_count > 0 else 30

    ballot_index = 0
    while ballot_index < total_ballots:
        if is_demo:
            draw_demo_watermark(c, width, height)
            draw_demo_header(c, width, height)

        for slot in range(ballots_per_page):
            if ballot_index >= total_ballots:
                break

            col = slot % 2
            row = slot // 2

            x = margin + col * (col_w + col_gap)
            ballot_top = height - margin - row * (ballot_h + row_gap)

            c.setStrokeColor(HexColor("#CCCCCC"))
            c.setDash(2, 2)
            c.rect(x, ballot_top - ballot_h, col_w, ballot_h)
            c.setDash()

            cx = x + col_w / 2
            y = ballot_top - 5 * mm

            c.setFillColor(NAVY)
            c.setFont("Helvetica-Bold", 9)
            c.drawCentredString(cx, y, election_name)
            y -= 4 * mm

            c.setFont("Helvetica", 7)
            c.setFillColor(HexColor("#666666"))
            c.drawCentredString(cx, y, f"Paper Ballot \u2014 Round {round_number}")
            y -= 4 * mm

            if round_number > 1:
                c.setFont("Helvetica-Bold", 6.5)
                c.setFillColor(HexColor("#C0392B"))
                c.drawCentredString(cx, y, "Vote ONLY for candidates announced by the chairman")
                y -= 4 * mm

            for item in office_data:
                office = item["office"]
                candidates = item["candidates"]

                c.setFillColor(NAVY)
                c.setFont("Helvetica-Bold", 7.5)
                max_sel = office.get("max_selections", office.get("vacancies", 1))
                c.drawString(x + 3 * mm, y, f"{office['name']} (select up to {max_sel})")
                y -= 4 * mm

                for cand in candidates:
                    c.setStrokeColor(NAVY)
                    c.setFillColor(HexColor("#FFFFFF"))
                    c.rect(x + 4 * mm, y - 0.5 * mm, 3 * mm, 3 * mm)

                    c.setFillColor(NAVY)
                    c.setFont("Helvetica", 7.5)
                    cand_name = cand.get("name", cand) if isinstance(cand, dict) else cand
                    c.drawString(x + 9 * mm, y, cand_name)
                    y -= 5 * mm

                y -= 1 * mm

            ballot_index += 1

        c.showPage()

    c.save()
    buf.seek(0)
    return buf


def generate_counter_sheet_pdf(election_name, congregation_name, offices_data,
                               member_count=0, is_demo=False):
    """
    Generate counter sheet PDF for paper ballot counting.

    Args:
        election_name: Election name string
        congregation_name: Full congregation name
        offices_data: List of dicts with 'office' (dict with name, max_selections)
                      and 'candidates' (list of dicts with name)
        member_count: Number of members for tick box count
        is_demo: If True, add DEMO watermark

    Returns:
        BytesIO buffer containing the PDF
    """
    max_votes = max(member_count, 50)

    buf = io.BytesIO()
    c = canvas.Canvas(buf, pagesize=A4)
    width, height = A4

    margin = 15 * mm
    box_size = 3.5 * mm
    box_gap = 0.8 * mm
    group_gap = 2.5 * mm
    group_size = 5
    tally_start = 55 * mm
    total_col = width - 25 * mm
    avail_width = total_col - tally_start - 5 * mm
    group_width = group_size * (box_size + box_gap) - box_gap + group_gap
    groups_per_row = int(avail_width / group_width)
    boxes_per_row = groups_per_row * group_size

    for office_item in offices_data:
        office = office_item["office"]
        candidates = office_item["candidates"]
        if not candidates:
            continue

        if is_demo:
            draw_demo_watermark(c, width, height)

        c.setFillColor(NAVY)
        c.setFont("Helvetica-Bold", 14)
        c.drawCentredString(width / 2, height - 18 * mm, "Paper Ballot Counting Sheet")
        c.setFont("Helvetica", 9)
        c.setFillColor(HexColor("#666666"))
        c.drawCentredString(width / 2, height - 25 * mm,
                            f"{congregation_name} \u2014 {election_name}")

        c.setFont("Helvetica", 8)
        c.drawString(margin, height - 33 * mm, "Counter: _________________________")
        c.drawString(width / 2, height - 33 * mm, "Date: _______________")

        max_sel = office.get("max_selections", office.get("vacancies", 1))
        c.setFillColor(NAVY)
        c.setFont("Helvetica-Bold", 12)
        c.drawString(margin, height - 42 * mm,
                     f"{office['name']} (select up to {max_sel})")

        y = height - 50 * mm

        for cand in candidates:
            row_height_est = box_size + box_gap + 1 * mm
            rows_needed = math.ceil(max_votes / boxes_per_row)
            candidate_height = 6 * mm + rows_needed * row_height_est + 8 * mm

            if y - candidate_height < 25 * mm:
                c.showPage()
                if is_demo:
                    draw_demo_watermark(c, width, height)
                y = height - 20 * mm

            cand_name = cand.get("name", cand) if isinstance(cand, dict) else cand
            c.setFillColor(NAVY)
            c.setFont("Helvetica-Bold", 10)
            c.drawString(margin, y, cand_name)

            c.setStrokeColor(NAVY)
            c.setFont("Helvetica", 7)
            c.setFillColor(HexColor("#666666"))
            c.drawString(total_col, y, "Total")
            c.rect(total_col, y - rows_needed * row_height_est - 2 * mm,
                   12 * mm, rows_needed * row_height_est + 2 * mm)

            y -= 5 * mm

            row_h = box_size + box_gap + 1 * mm
            c.setStrokeColor(HexColor("#CCCCCC"))
            boxes_drawn = 0
            row = 0
            while boxes_drawn < max_votes:
                for g_idx in range(groups_per_row):
                    if boxes_drawn >= max_votes:
                        break
                    gx = tally_start + g_idx * group_width
                    by = y - row * row_h
                    for b in range(group_size):
                        if boxes_drawn >= max_votes:
                            break
                        bx = gx + b * (box_size + box_gap)
                        c.rect(bx, by, box_size, box_size)
                        boxes_drawn += 1

                    if boxes_drawn > 0 and boxes_drawn % group_size == 0:
                        c.setFont("Helvetica", 4.5)
                        c.setFillColor(HexColor("#BBBBBB"))
                        label_x = gx + (group_size * (box_size + box_gap) - box_gap) / 2
                        c.drawCentredString(label_x, by - 2.5 * mm, str(boxes_drawn))
                        c.setFillColor(NAVY)
                        c.setStrokeColor(HexColor("#CCCCCC"))

                row += 1

            y -= row * row_h + 4 * mm

        c.setFont("Helvetica", 7)
        c.setFillColor(HexColor("#999999"))
        c.drawCentredString(width / 2, 12 * mm,
                            "Compare both counter sheets. Totals must match before entering into the app.")

        c.showPage()

    c.save()
    buf.seek(0)
    return buf


def generate_results_pdf(election_name, rounds_data, is_demo=False):
    """
    Generate election results PDF.

    Args:
        election_name: Election name string
        rounds_data: List of dicts per round, each containing:
            - round_number: int
            - used_codes: int (digital votes cast)
            - offices: list of dicts with:
                - name: office name
                - candidates: list of dicts with name, digital, paper, total
        is_demo: If True, add DEMO watermark

    Returns:
        BytesIO buffer containing the PDF
    """
    from datetime import datetime

    buf = io.BytesIO()

    def on_page(canvas_obj, doc):
        if is_demo:
            draw_demo_watermark(canvas_obj, A4[0], A4[1])
            draw_demo_header(canvas_obj, A4[0], A4[1])

    doc = SimpleDocTemplate(buf, pagesize=A4, topMargin=20 * mm, bottomMargin=20 * mm)
    styles = getSampleStyleSheet()
    elements = []

    title_style = styles["Title"]
    title_style.textColor = NAVY
    elements.append(Paragraph(election_name, title_style))
    elements.append(Paragraph(
        f"Results as at {datetime.now().strftime('%d %B %Y %H:%M')}", styles["Normal"]
    ))
    elements.append(Spacer(1, 10 * mm))

    for rd in rounds_data:
        elements.append(Paragraph(f"Round {rd['round_number']}", styles["Heading2"]))
        elements.append(Paragraph(f"Digital votes cast: {rd['used_codes']}", styles["Normal"]))
        elements.append(Spacer(1, 5 * mm))

        for office in rd["offices"]:
            elements.append(Paragraph(office["name"], styles["Heading3"]))

            table_data = [["Candidate", "Digital", "Paper", "Total"]]
            for cand in office["candidates"]:
                table_data.append([
                    cand["name"], str(cand["digital"]),
                    str(cand["paper"]), str(cand["total"])
                ])

            table = Table(table_data, colWidths=[200, 70, 70, 70])
            table.setStyle(TableStyle([
                ("BACKGROUND", (0, 0), (-1, 0), NAVY),
                ("TEXTCOLOR", (0, 0), (-1, 0), colors.white),
                ("ALIGN", (1, 0), (-1, -1), "CENTER"),
                ("FONTNAME", (0, 0), (-1, 0), "Helvetica-Bold"),
                ("FONTSIZE", (0, 0), (-1, -1), 10),
                ("BOTTOMPADDING", (0, 0), (-1, 0), 8),
                ("TOPPADDING", (0, 0), (-1, 0), 8),
                ("GRID", (0, 0), (-1, -1), 0.5, colors.grey),
                ("ROWBACKGROUNDS", (0, 1), (-1, -1),
                 [colors.white, HexColor("#F5F5F5")]),
            ]))
            elements.append(table)
            elements.append(Spacer(1, 8 * mm))

    doc.build(elements, onFirstPage=on_page, onLaterPages=on_page)
    buf.seek(0)
    return buf

        if not valid_code:

@app.route("/admin/election/<int:election_id>/dual-ballot-pdf")
@admin_required
def admin_dual_ballot_pdf(election_id):
    """Generate dual ballot handout PDF (demo mode only)."""
    if get_setting("is_demo", "0") != "1":
        abort(403)

    db = get_db()
    election = db.execute("SELECT * FROM elections WHERE id = ?", (election_id,)).fetchone()
    if not election:
        abort(404)

    offices = db.execute(
        "SELECT * FROM offices WHERE election_id = ? ORDER BY sort_order", (election_id,)
    ).fetchall()

    office_data = []
    for office in offices:
        candidates = db.execute(
            "SELECT * FROM candidates WHERE office_id = ? AND active = 1 ORDER BY name",
            (office["id"],)
        ).fetchall()
        office_data.append({"office": dict(office), "candidates": [dict(c) for c in candidates]})

    # Get codes — we need plaintext codes, but they're only stored as hashes.
    # For the dual ballot, we need to regenerate or use session codes.
    # Since this is demo mode, generate fresh codes if needed.
    session_key = f"codes_{election_id}"
    codes = session.get(session_key)

    if not codes:
        # In demo mode, generate new plaintext codes for the PDF
        # First, delete existing codes and regenerate
        existing_count = db.execute(
            "SELECT COUNT(*) FROM codes WHERE election_id = ?", (election_id,)
        ).fetchone()[0]
        if existing_count > 0:
            # Can't get plaintext back — need to regenerate
            db.execute("DELETE FROM codes WHERE election_id = ?", (election_id,))
            db.commit()
        codes = generate_codes(election_id, 20)
        session[session_key] = codes

    buf = generate_dual_ballot_handout_pdf(
        election_name=election["name"],
        congregation_name=get_setting("congregation_name", "Free Reformed Church"),
        office_data=office_data,
        codes=codes[:20],
        wifi_ssid=get_setting("wifi_ssid", "ChurchVote"),
        wifi_password=get_setting("wifi_password", ""),
        base_url=get_setting("voting_base_url", "http://192.168.8.100:5000"),
    )

    return send_file(buf, mimetype="application/pdf", as_attachment=True,
                     download_name="demo_dual_ballot_handout.pdf")
```
